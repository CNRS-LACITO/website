Fig. 2: DTD (document type definition)

<!-- DTD des documents d'archive (25/05/1999)			      -->





<!ELEMENT ARCHIVE	(TEXT)+						>





<!ELEMENT TEXT		(HEADER,BODY)					>





<!ELEMENT HEADER	(IMAGE?,TITLE,SOUNDFILE)			>


<!ELEMENT TITLE		(#PCDATA)					>


<!ELEMENT SOUNDFILE	EMPTY						>





<!ELEMENT BODY		(S*)						>


<!ELEMENT S		(AUDIO,TRANSCR,TRADUC+)				>


<!ELEMENT AUDIO		EMPTY						>


<!ELEMENT TRADUC	(#PCDATA)					>


<!ELEMENT TRANSCR	(W|PONCT)*					>


<!ELEMENT W		(FORM,GLS?)					>


<!ELEMENT FORM		(#PCDATA|FOREIGN)*				>


<!ELEMENT PONCT		(#PCDATA)					>


<!ELEMENT GLS		(#PCDATA)					>


<!ELEMENT FOREIGN	(#PCDATA)					>





<!ELEMENT IMAGE		(#PCDATA)					>





<!ATTLIST IMAGE		src   CDATA      #REQUIRED


			href  CDATA      #REQUIRED    			>


<!ATTLIST AUDIO		start CDATA      #REQUIRED


			end   CDATA      #REQUIRED    			>


<!ATTLIST BODY		lang  CDATA      #REQUIRED    			>


<!ATTLIST FOREIGN	lang  CDATA      #REQUIRED    			>


<!ATTLIST TRADUC	lang  CDATA      "Francais"   			>


<!ATTLIST TRANSCR	lang  CDATA      #IMPLIED     			>


<!ATTLIST S		id    ID         #REQUIRED


			who   CDATA      #IMPLIED			>


<!ATTLIST SOUNDFILE	href  CDATA      #REQUIRED			>
